Write a program for ddition of two numbers.

class Addition
{
  int x, y;
 
public:
  void input() {
    cout << "Input two integers\n";
    cin >> x >> y;
  }
 
  void add() {
    cout << "Result = " << x + y;
  }
 
};
 
int main()
{
   Addition m; // Creating object of class
 
   m.input(); // calling method of class
   m.add();   //calling method of class
 
   return 0;
}
